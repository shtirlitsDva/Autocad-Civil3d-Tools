The persisted model (C#)

        
public sealed class LabelTheme
{
    public LabelStyle   Style    { get; set; }   // Glass | Blueprint
                                                 // | Segmented | Fluent
                                                 // | Terminal
    public SegmentMode  TileMode { get; set; }   // Flush|Pop|Bordered
    public ThemeColors  Colors   { get; set; }
    public SeriesStyle  Series   { get; set; }
    public FontSet      Fonts    { get; set; }
    public int          Padding  { get; set; }   // pt, 4..14
    public bool         Rounded  { get; set; }
}

public sealed class ThemeColors
{
    public string Frame, PanelTop, PanelBottom,
                  Id, Body, Divider, Background;
}

public sealed class SeriesStyle
{
    public bool   Show     { get; set; }
    public ChipFill Fill   { get; set; }   // Filled|Outline
    public ChipPos  Pos    { get; set; }   // After|Before
    public string[] Colors { get; set; }   // [I, II, III]
}

Note on fonts

      
Graphviz HTML‑labels render with whatever font the FACE names — it must be installed on the host that runs the export. Ship the theme with safe fallbacks (e.g. Courier New, Segoe UI) or bundle the fonts with the viewer. Everything else here (gradients, rounded tables, per‑side borders, colored chips) is pure Graphviz and needs no external assets.
